import { Canvas, Meta, Story, Source } from '@storybook/addon-docs'

# Binary True / false

<Canvas>
	<Story
		name="True/false"
		play={(ctx) =>
			checkTest({
				...ctx,
				functions: [booleanTest]
			})
		}
		args={{
			id: 4,
			label: 'Accept terms and conditions',
			name: 'ck4',
			trueValue: true,
			falseValue: false
		}}>
		{(args) => {
			return {
				components: { VvCheckbox },
				setup() {
					return { args }
				},
				data() {
					return {
						checkValue: false
					}
				},
				template: `<div :data-testData="checkValue" :key="checkValue">
						      <vv-checkbox v-model="checkValue" v-bind="args"  />
							  <div>Selected value: {{checkValue}}</div>
						   </div>`
			}
		}}
	</Story>
</Canvas>
