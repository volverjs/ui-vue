# Rounded

A VvButtonGroup with the <strong>rounded</strong> prop will be render with <strong>.vv-button-group--rounded</strong> css class. <br/>

<Source language="jsx" code={'<vv-button rounded></vv-button>'} />

<Canvas>
	<Story name="rounded">
		{() => {
			return {
				components: { VvButton, VvButtonGroup },
				template:
					/*html*/
					`
                        <vv-button-group rounded>
                            <vv-button>Button 1</vv-button>
                            <vv-button>Button 2</vv-button>
                            <vv-button>Button 3</vv-button>
                        </vv-button-group>
                    `
			}
		}}
	</Story>
</Canvas>

# Mix rounded buttons

You can manage the rounded stile of single buttons into group. Set rounded=false and add <strong>.vv-button--rounded</strong> to single buttons.

<Canvas>
	<Story name="mix">
		{() => {
			return {
				components: { VvButton, VvButtonGroup },
				template:
					/*html*/
					`
					<div class="flex flex-col">
                        <vv-button-group>
                            <vv-button class="vv-button vv-button--rounded">Button 1</vv-button>
                            <vv-button>Button 2</vv-button>
                            <vv-button class="vv-button vv-button--rounded">Button 3</vv-button>
                        </vv-button-group>
						 <vv-button-group class="mt-22">
                            <vv-button>Button 1</vv-button>
                            <vv-button>Button 2</vv-button>
                            <vv-button class="vv-button vv-button--rounded">Button 3</vv-button>
                        </vv-button-group>
					</div>
                    `
			}
		}}
	</Story>
</Canvas>
