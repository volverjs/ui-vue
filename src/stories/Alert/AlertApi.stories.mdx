import { Meta, Story } from '@storybook/addon-docs'
import VvAlert from '../../components/VvAlert/VvAlert.vue'

import { useNotify } from '@/composables/notify/useNotify'

<Meta title="Components/Alert/Api" />

# Api

<Story
	name="AlertApi"
	args={{ icon: 'bell' }}>
	{
		(args, { argTypes }) => ({
			props: Object.keys(argTypes),
			setup() {
				const {messages, addMessage} = useNotify()
				
				function openAlert(position){
					addMessage({
						type: 'alert', 
						title: `Alert ${Math.random()}`,
						top: position === 'top',
						bottom: position === 'bottom',
						left: true, 
						visible: true,
						closable: true
					})
    			}

    			return { args, openAlert }
    		},
    		template:
    			/**html */
    			`
    			<button class="vv-button" @click="openAlert('top')">Aggiungi un alert top</button>
    			<button class="vv-button" @click="openAlert('bottom')">Aggiungi un alert bottom</button>
    			`
    	})
    }

</Story>
